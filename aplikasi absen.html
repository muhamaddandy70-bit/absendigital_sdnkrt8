    <script>

        // --- KONFIGURASI APLIKASI ---

        let appSettings = {};

        

        // --- FUNGSI SIMPAN KE GOOGLE SHEET (APPS SCRIPT) ---

        async function saveDataToGoogleSheet(dataAbsen) {

            const url = appSettings.appsScriptUrl;

            if (!url) {

                document.getElementById('sync_status').textContent = '⚠️ URL Apps Script belum diatur.';

                return;

            }

            document.getElementById('sync_status').textContent = `Menyimpan data ${dataAbsen.nama}...`;


            try {

                const response = await fetch(url, {

                    method: 'POST',

                    mode: 'no-cors', // Diperlukan untuk Apps Script Web App

                    cache: 'no-cache',

                    headers: {

                        'Content-Type': 'application/json'

                    },

        function downloadCSV(csvContent, fileName) {

            const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            link.setAttribute("href", URL.createObjectURL(blob));

            link.setAttribute("download", `${fileName}.csv`);

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        }

        function renderRekapBulanan() {

            const rekapBulananContainer = document.getElementById('rekap-bulanan-container');

            rekapBulananContainer.innerHTML = `<p class="text-gray-500 text-center py-4">Memproses data, mohon tunggu...</p>`;

            const bulan = parseInt(document.getElementById('pilih-bulan').value);

            const tahun = parseInt(document.getElementById('pilih-tahun').value);

            const daysInMonth = new Date(tahun, bulan + 1, 0).getDate();

            const dataBulanan = {};

            for (let day = 1; day <= daysInMonth; day++) {

                const date = new Date(tahun, bulan, day);

                const key = `absensi_${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;

                const dataHarian = JSON.parse(localStorage.getItem(key)) || [];

                dataHarian.forEach(absen => {

                    if (!dataBulanan[absen.nama]) dataBulanan[absen.nama] = {};

                    dataBulanan[absen.nama][day] = absen.status.charAt(0);

                });

            }

            let tableHTML = `<table class="min-w-full divide-y divide-gray-200 border border-gray-300"><thead class="bg-gray-100"><tr><th class="border border-gray-300 px-2 py-2 text-sm font-semibold">Nama Siswa</th>`;

            for (let day = 1; day <= daysInMonth; day++) { tableHTML += `<th class="border border-gray-300 px-2 py-2 text-sm font-semibold w-10">${day}</th>`; }

            tableHTML += `<th class="border border-gray-300 px-2 py-2 text-sm font-semibold bg-green-100">H</th><th class="border border-gray-300 px-2 py-2 text-sm font-semibold bg-yellow-100">S</th><th class="border border-gray-300 px-2 py-2 text-sm font-semibold bg-blue-100">I</th><th class="border border-gray-300 px-2 py-2 text-sm font-semibold bg-red-100">A</th></tr></thead><tbody class="bg-white divide-y divide-gray-200">`;

            daftarSiswa.forEach(siswa => {

                tableHTML += `<tr><td class="border border-gray-300 px-2 py-2 text-sm font-medium whitespace-nowrap">${siswa.nama}</td>`;

                let counts = { H: 0, S: 0, I: 0, A: 0 };

                for (let day = 1; day <= daysInMonth; day++) {

                    const status = dataBulanan[siswa.nama]?.[day] || 'A';

                    if(status !== 'A') counts[status]++;

                    tableHTML += `<td class="border border-gray-300 status-cell status-${status}">${status}</td>`;

                }

                counts.A = daysInMonth - (counts.H + counts.S + counts.I);

                tableHTML += `<td class="border border-gray-300 text-center font-bold text-sm py-2">${counts.H}</td><td class="border border-gray-300 text-center font-bold text-sm py-2">${counts.S}</td><td class="border border-gray-300 text-center font-bold text-sm py-2">${counts.I}</td><td class="border border-gray-300 text-center font-bold text-sm py-2">${counts.A}</td></tr>`;

            });

            tableHTML += `</tbody></table>`;

            rekapBulananContainer.innerHTML = tableHTML;

        }

                });

            for (let day = 1; day <= daysInMonth; day++) csvHeader.push(day);

            csvHeader.push("Total Hadir", "Total Sakit", "Total Izin", "Total Alfa");

            let csvContent = csvHeader.join(',') + '\n';

            daftarSiswa.forEach(siswa => {

                let row = [`"${siswa.nama}"`];

                let counts = { H: 0, S: 0, I: 0, A: 0 };

                for (let day = 1; day <= daysInMonth; day++) {

                    const status = dataBulanan[siswa.nama]?.[day] || 'A';

                    if(status !== 'A') counts[status]++;

                    row.push(status);

                }

                counts.A = daysInMonth - (counts.H + counts.S + counts.I);

            });
